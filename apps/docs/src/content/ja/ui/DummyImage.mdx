import { DummyImage } from '@lism-css/ui/astro';

`https://cdn.lism-css.com/`からダミーのプレースホルダー画像を出力します。

## Import

```jsx
import { DummyImage } from '@lism-css/ui/astro';
// or
import { DummyImage } from '@lism-css/ui/react';
```
