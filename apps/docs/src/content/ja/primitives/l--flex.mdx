### flexプロパティの指定

<Preview>
  <PreviewTitle>flexプロパティの使用例</PreviewTitle>
  <PreviewArea resize p="30">
    <Flex g="20" fxw="wrap" jc={['baseline', 'center']} className="has--itemGuide">
      <div>Flex Content</div>
      <div>Flex Content</div>
      <div>Flex Content</div>
    </Flex>
  </PreviewArea>
  <PreviewCode slot="tab" label="JSX">
  ```jsx
  <Flex fxw="wrap" jc="center" g="20">
    <div>Flex Content</div>
    <div>Flex Content</div>
    <div>Flex Content</div>
  </Flex>
  ```
  </PreviewCode>
  <PreviewCode slot="tab" label="HTML">
  ```html
  <div class="l--flex -fxw:wrap -g:20 -jc:center">
    <div>Flex Content</div>
    <div>Flex Content</div>
    <div>Flex Content</div>
  </div>
  ```
  </PreviewCode>
</Preview>
