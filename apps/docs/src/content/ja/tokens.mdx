import { Box, Stack, Cluster, Columns, Frame, Center, Inline, Text } from 'lism-css/astro';
import { DummyText } from '@lism-css/ui/astro';

export const ColorBadge = ({color, label}) => {
  return <Cluster g='10'>
      <Inline w='1.5em' ar='1/1' bgc={color} bdrs='10' bxsh='10' />
      <code>{label || color}</code>
    </Cluster>;
}


Lism CSS では、主要なCSSプロパティに対してデザイントークンを用意しています。


## [COLOR](/docs/tokens/colors/)

|トークン名|変数|
|---|---|
|<ColorBadge color='base' />|`--base`|
|<ColorBadge color='base-2' />|`--base-2`|
|<ColorBadge color='text' />|`--text`|
|<ColorBadge color='text-2' />|`--text-2`|
|<ColorBadge color='divider' />|`--divider`|
|<ColorBadge color='link' />|`--link`|
|<ColorBadge color='brand' />|`--brand`|
|<ColorBadge color='accent' />|`--accent`|

パレットカラー（`--red`, `--blue`, `--green` 等）やカスタマイズ方法は [カラートークン](/docs/tokens/colors/) を参照してください。


## [Typography](/docs/tokens/typography/)

|トークン|変数|
|---|---|
|`FF` (font-family)|`--ff--base`, `--ff--accent`, `--ff--mono`|
|`FW` (font-weight)|`--fw--light`, `--fw--normal`, `--fw--bold`|
|`FZ` (font-size)|`--fz--2xs` 〜 `--fz--5xl`|
|`LH` (line-height)|`--hl--xs`, `--hl--s`, `--hl--base`, `--hl--l`|
|`LTS` (letter-spacing)|`--lts--s`, `--lts--base`, `--lts--l`|

各トークンの詳細やスケーリング規則は [タイポグラフィトークン](/docs/tokens/typography/) を参照してください。

## `SHADOW`


{/* <Callout type='warning'>
box-shadow は少し特殊で、ユーティリティクラス内で複数のshadow用変数が組み合わされています。
</Callout> */}

|トークン値|変数|クラス|
|---|---|---|
|`10`|`--bxsh--10`|`-bxsh:10`|
|`20`|`--bxsh--20`|`-bxsh:20`|
|`30`|`--bxsh--30`|`-bxsh:30`|
|`40`|`--bxsh--40`|`-bxsh:40`|

`--bxsh--{token}`の値の中身は、次の変数で構成されます。

|構成変数|初期値|
|---|---|
|`--shc`|`hsl(220 4% 8% / 5%)`|
|`--shsz--5`|`0px 1px 3px`|
|`--shsz--10`|`0px 2px 6px`|
|`--shsz--20`|`0px 4px 12px`|
|`--shsz--30`|`0px 8px 24px`|
|`--shsz--40`|`0px 16px 48px`|



{/* `bxsh`は、`1` ~ `4`, `-1` ~ `-4` までの値に対してユーティリティクラスを用意していますが、内部では2段階のshadowを組み合わせることでより立体的で美しいシャドウを表現できるようになっています。 */}

## `SIZE`

コンテンツサイズ用のトークンです。

|トークン値|変数|
|---|---|
|`xs`|`--sz--xs`|
|`s`|`--sz--s`|
|`m`|`--sz--m`|
|`l`|`--sz--l`|
|`xl`|`--sz--xl`|
|`min`|`--sz--min`|

`max-sz`, <ModLink slug="is--wrapper" label=".is--wrapper" /> 等で使用できます。
