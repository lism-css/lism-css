import { Lism, Grid, Frame, Box, Flex, Spacer, Icon } from 'lism-css/astro';
import { DummyText } from '@lism-css/ui/astro';
import { HandSwipeLeftIcon, MouseScrollIcon } from '@phosphor-icons/react';

export const SwipeHelp = ({icon}) => <span> ( <Icon icon={icon || HandSwipeLeftIcon} /> <small>Scroll to preview</small> )</span>;

export const ReelImages = ({ length=6, hasID, isV, ...props }) => {
  return (
    <>
      {Array.from({length}, (v, i) => i).map((i) => {
        let imgNum = i + 1; //(i + 1) % 3 || 3;
        const src = 'https://cdn.lism-css.com/img/u-' + imgNum + '.jpg';
        return <Frame id={hasID ? `image-${i+1}` : ''} ar="16/9" key={i} {...props}>
          <img src={src} alt="" loading="lazy" width="960" height="640" />
        </Frame>
      })}
    </>
  );
}

Examples of creating a reel component that displays content with horizontal scrolling.

## Disabling the reel at larger screen sizes

To show the reel only on small screens, set `gaf` (`grid-auto-flow`) back to `row`.
The following example demonstrates switching to a 2-column layout by setting `gtc` (`grid-template-columns`) to `repeat(2, 1fr)` at the same breakpoint.

<Preview>
  <PreviewTitle>Example: disable reel only at `~@md`<SwipeHelp /></PreviewTitle>
  <PreviewArea resize p="10">
    <Grid hasSnap ov="auto" g="15" gaf={['column', null, 'row']} gac="75%" gtc={[null, null, 'repeat(2, 1fr)']}>
      <ReelImages />
    </Grid>
  </PreviewArea>
  <PreviewCode>
    ```jsx "gaf" "gtc"
    <Grid hasSnap ov="auto" g="15" gaf={['column', null, 'row']} gac="75%" gtc={[null, null, 'repeat(2, 1fr)']}>
      ...
    </Grid>
    ```
  </PreviewCode>
</Preview>
