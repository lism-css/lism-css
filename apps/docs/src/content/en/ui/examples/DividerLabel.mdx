import { Divider, Flex, Box, Stack, Inline } from 'lism-css/astro';


Examples of creating a divider component with a label text.


## Examples

<Preview>
  <PreviewTitle>Example</PreviewTitle>
  <PreviewArea p="20" >
    <Box bgc="base-2" c="text-2" py="40" ta="center">...Contents...</Box>
    <Flex ai="center" g="20" lh="s">
      <Divider fx="1" />
      <Inline fz="s">Label Text</Inline>
      <Divider fx="1" />
    </Flex>
    <Box bgc="base-2" c="text-2" py="40" ta="center">...Contents...</Box>
  </PreviewArea>
  <PreviewCode slot="tab" label="JSX">
    ```jsx {2-6}
    <Box bgc="base-2" c="text-2" py="40" ta="center">...Contents...</Box>
    <Flex ai="center" g="20" lh="s">
      <Divider fx="1" />
      <Inline fz="s">Label Text</Inline>
      <Divider fx="1" />
    </Flex>
    <Box bgc="base-2" c="text-2" py="40" ta="center">...Contents...</Box>
    ```
  </PreviewCode>
  <PreviewCode slot="tab" label="HTML">
    ```html
    <div class="l--box -bgc:base-2 -c:text-2 -py:40 -ta:center">…Contents…</div>

      <div class="a--divider -fx:1" aria-hidden="true"></div>
      <span class="-fz:s">Label Text</span>

    </div>

    ```

### Vertical orientation

<Preview>
  <PreviewTitle>Changing `flex-direction`</PreviewTitle>
  <PreviewArea p="20" layout="">
    <Flex ai="center" g="15">
      <Box fx="1" bgc="base-2" py="40" ta="center">Box</Box>
      <Stack ai="center" g="10" lh="s" fz="s" aslf="stretch">
        <Divider fx="1" wm="vertical-rl" />
        <span>Label</span>
        <Divider fx="1" wm="vertical-rl" />
      </Stack>
      <Box fx="1" bgc="base-2" py="40" ta="center">Box</Box>
    </Flex>
  </PreviewArea>
  <PreviewCode slot="tab" label="JSX">
    ```jsx 'wm="vertical-rl"'
    <Flex ai="center" g="15">
      <Box fx="1" bgc="base-2" py="40" ta="center">
        Box
      </Box>
      <Stack ai="center" g="10" lh="s" fz="s" aslf="stretch">
        <Divider fx="1" wm="vertical-rl" />
        <span>Label</span>
        <Divider fx="1" wm="vertical-rl" />
      </Stack>
      <Box fx="1" bgc="base-2" py="40" ta="center">
        Box
      </Box>
    </Flex>
    ```
  </PreviewCode>
  <PreviewCode slot="tab" label="HTML">
    ```html
    <div class="l--flex -ai:center -g:15">
      <div class="l--box -fx:1 -bgc:base-2 -py:40 -ta:center">Box</div>
      <div class="l--stack -ai:center -g:10 -lh:s -fz:s -aslf:stretch">
        <div class="a--divider -wm:vertical-rl -fx:1" aria-hidden="true"></div>
        <span>Label</span>
        <div class="a--divider -wm:vertical-rl -fx:1" aria-hidden="true"></div>
      </div>
      <div class="l--box -fx:1 -bgc:base-2 -py:40 -ta:center">Box</div>
    </div>
    ```
  </PreviewCode>
